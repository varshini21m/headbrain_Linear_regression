#mean of x and y
mean_x = np.mean(X)
mean_y = np.mean(Y)

# total number of values
n = len(X)

# using the formula to cal b1:is mean and b0:is C intercept 
# m = sum((x-x')(y-y'))
#     __________________
#       sum((x-x')^2)

numer = 0
denom = 0
for i in range(n):
    numer += (X[i] - mean_x)*(Y[i] - mean_y)
    denom += (X[i] - mean_x)**2
b1 = numer / denom
b0 = mean_y - (b1 * mean_x)
# y = mx + c now here became b0 = b1x + c

print(b1,b0)
